POST /api/schematics/upload

{{ T .Language "Upload a .nbt schematic file. Returns a preview token. The schematic is not published until you complete the publish flow via the web UI." }}

{{ T .Language "Parameters" }}

{{ T .Language "Send as" }} multipart/form-data

{{ T .Language "Field" }}{{ T .Language "Type" }}{{ T .Language "Description" }}
file / nbtfile{{ T .Language "Required" }}{{ T .Language "The .nbt schematic file (max 10MB)" }}

{{ T .Language "Code Samples" }}

curl -X POST \
  -H "X-API-Key: YOUR_KEY" \
  -F "file=@my_schematic.nbt" \
  "https://createmod.com/api/schematics/upload"
const form = new FormData();
form.append("file", fileInput.files[0]);

const resp = await fetch("https://createmod.com/api/schematics/upload", {
  method: "POST",
  headers: { "X-API-Key": "YOUR_KEY" },
  body: form
});
const data = await resp.json();
console.log(data.token, data.url);
import requests

with open("my_schematic.nbt", "rb") as f:
    resp = requests.post(
        "https://createmod.com/api/schematics/upload",
        files={"file": f},
        headers={"X-API-Key": "YOUR_KEY"}
    )
data = resp.json()
print(data["token"], data["url"])
body := &bytes.Buffer{}
writer := multipart.NewWriter(body)
part, _ := writer.CreateFormFile("file", "my_schematic.nbt")
io.Copy(part, fileReader)
writer.Close()

req, _ := http.NewRequest("POST",
    "https://createmod.com/api/schematics/upload", body)
req.Header.Set("X-API-Key", "YOUR_KEY")
req.Header.Set("Content-Type", writer.FormDataContentType())
resp, _ := http.DefaultClient.Do(req)
